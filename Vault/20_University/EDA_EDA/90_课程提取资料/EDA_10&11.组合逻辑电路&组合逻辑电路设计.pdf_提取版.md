module sy(sel,a,b,c,d,y);
input [1:0]sel;//两位的选择信号
input a,b,c,d;
output reg y;
always@(sel,a,b,c,d);
if(sel==2b'00) y=a;
else if(sel==2b'01) y=b;
else if(sel==2b'10) y=c;
else y=d;
end
endmodule
module sy(sel,a,b,c,d,y);
input [1:0]sel;//两位的选择信号
input a,b,c,d;
output reg y;
always@(sel,a,b,c,d);
case(sel)
2b'00: y=a;
2b'01: y=b;
2b'10: y=c;
default: y=d;
end
endmodule
module sy(a,even_bit,odd_bit);
input [7:0]a;
output wire even_bit;
output wire odd_bit;
assign even_bit=^a;(当a中有奇数个 1 时，异或的结果为 1 ）
assign odd_bit=~even_bit;
endmodule
