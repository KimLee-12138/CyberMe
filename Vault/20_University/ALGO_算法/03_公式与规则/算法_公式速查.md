# 第一部分：公式定理

## 一、主定理

对于 $T(n) = aT(n/b) + f(n)$：

| Case | 条件 | 结论 |
|------|------|------|
| 1 | $f(n) = O(n^{\log_b a - \epsilon})$ | $T(n) = \Theta(n^{\log_b a})$ |
| 2 | $f(n) = \Theta(n^{\log_b a})$ | $T(n) = \Theta(n^{\log_b a}\log n)$ |
| 3 | $f(n) = \Omega(n^{\log_b a + \epsilon})$ + 正则条件 | $T(n) = \Theta(f(n))$ |

**口诀**：比较 $n^{\log_b a}$ 与 $f(n)$，谁大谁主导。

## 二、常见递归式及解

| 递归式 | 解 | 对应算法 |
|--------|-----|---------|
| $T(n) = T(n/2) + O(1)$ | $O(\log n)$ | 二分查找 |
| $T(n) = 2T(n/2) + O(n)$ | $O(n\log n)$ | 归并排序、最近点对 |
| $T(n) = T(n-1) + O(n)$ | $O(n^2)$ | 选择排序、快排最坏 |
| $T(n) = 3T(n/2) + O(n)$ | $O(n^{1.585})$ | Karatsuba |
| $T(n) = 7T(n/2) + O(n^2)$ | $O(n^{2.807})$ | Strassen |
| $T(n) \le T(n/5) + T(7n/10) + O(n)$ | $O(n)$ | 线性时间选择 |

## 三、DP 经典状态转移方程

**LCS**：
$$L[i][j] = \begin{cases} L[i-1][j-1]+1 & X_i=Y_j \\ \max(L[i-1][j], L[i][j-1]) & X_i \neq Y_j \end{cases}$$

**矩阵链相乘**：
$$m[i][j] = \min_{i \le k < j}\{ m[i][k] + m[k+1][j] + p_{i-1}p_kp_j \}$$

**Floyd**：
$$D_k[i][j] = \min(D_{k-1}[i][j], \; D_{k-1}[i][k] + D_{k-1}[k][j])$$

**0-1 背包**：
$$V[i][j] = \begin{cases} V[i-1][j] & j < s_i \\ \max(V[i-1][j], \; V[i-1][j-s_i] + v_i) & j \ge s_i \end{cases}$$

## 四、排序算法复杂度速查

| 算法 | 最好 | 平均 | 最坏 | 空间 | 稳定 |
|------|------|------|------|------|------|
| 冒泡 | $O(n)$ | $O(n^2)$ | $O(n^2)$ | $O(1)$ | ✓ |
| 选择 | $O(n^2)$ | $O(n^2)$ | $O(n^2)$ | $O(1)$ | ✗ |
| 插入 | $O(n)$ | $O(n^2)$ | $O(n^2)$ | $O(1)$ | ✓ |
| 归并 | $O(n\log n)$ | $O(n\log n)$ | $O(n\log n)$ | $O(n)$ | ✓ |
| 快排 | $O(n\log n)$ | $O(n\log n)$ | $O(n^2)$ | $O(\log n)$ | ✗ |
| 堆排 | $O(n\log n)$ | $O(n\log n)$ | $O(n\log n)$ | $O(1)$ | ✗ |
| 基数 | $O(dn)$ | $O(dn)$ | $O(dn)$ | $O(n+k)$ | ✓ |

## 五、图算法复杂度速查

| 算法 | 用途 | 复杂度 |
|------|------|--------|
| BFS / DFS | 遍历 | $O(V+E)$ |
| 拓扑排序 | DAG 排序 | $O(V+E)$ |

# 第二部分：代码模板

## 模板 1：二分查找

```
int binarySearch(int A[], int n, int x) {
    int low = 0, high = n - 1;
    while (low <= high) {
        int mid = low + (high - low) / 2;   // 防溢出
        if (A[mid] == x) return mid;
        else if (A[mid] < x) low = mid + 1;
        else high = mid - 1;
    }
    return -1;
}
```

**变体**：查找第一个 ≥ x 的位置 → `A[mid] >= x` 时收缩右边界。

## 模板 2：归并排序

```
void mergeSort(int A[], int left, int right) {
    if (left >= right) return;
    int mid = left + (right - left) / 2;
    mergeSort(A, left, mid);
    mergeSort(A, mid + 1, right);
    merge(A, left, mid, right);
}

void merge(int A[], int left, int mid, int right) {
    int n1 = mid - left + 1, n2 = right - mid;
    int L[n1], R[n2];
    for (int i = 0; i < n1; i++) L[i] = A[left + i];
    for (int j = 0; j < n2; j++) R[j] = A[mid + 1 + j];
    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2)
        A[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];
    while (i < n1) A[k++] = L[i++];
    while (j < n2) A[k++] = R[j++];
}
```

## 模板 3：快速排序

```
void quickSort(int A[], int low, int high) {
    if (low >= high) return;
    int p = partition(A, low, high);
    quickSort(A, low, p - 1);
    quickSort(A, p + 1, high);
}

int partition(int A[], int low, int high) {
    int pivot = A[low];
    int i = low, j = high;
    while (i < j) {
        while (i < j && A[j] >= pivot) j--;   // 先从右往左
        while (i < j && A[i] <= pivot) i++;   // 再从左往右
        if (i < j) swap(A[i], A[j]);
    }
    swap(A[low], A[i]);
    return i;
}
```

**注意**：必须先从右向左扫描！

## 模板 4：LCS（最长公共子序列）

```
int LCS(string X, string Y) {
    int m = X.size(), n = Y.size();
    vector<vector<int>> dp(m+1, vector<int>(n+1, 0));
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (X[i-1] == Y[j-1])
                dp[i][j] = dp[i-1][j-1] + 1;
            else
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
        }
    }
    return dp[m][n];
}
```

**空间优化**：只需保留上一行，降为 $O(n)$。

## 模板 5：0-1 背包

```
int knapsack(int C, int w[], int v[], int n) {
    vector<int> dp(C + 1, 0);
    for (int i = 0; i < n; i++) {
        for (int j = C; j >= w[i]; j--) {   // 逆序！
            dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
        }
    }
    return dp[C];
}
```

**关键**：$j$ 必须**逆序遍历**，否则每个物品会被重复使用。

## 模板 6：Floyd 全源最短路径

```
void floyd(vector<vector<int>>& D, int n) {
    for (int k = 0; k < n; k++)           // k 在最外层！
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                D[i][j] = min(D[i][j], D[i][k] + D[k][j]);
}
```

**注意**：$k$ 循环位置不可改变，三层顺序固定。

## 模板 7：Dijkstra（堆优化）

```
vector<int> dijkstra(vector<vector<pair<int,int>>>& adj, int s) {
    int n = adj.size();
    vector<int> dist(n, INT_MAX);
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    dist[s] = 0; pq.push({0, s});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;     // 懒惰删除
        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}
```

## 模板 8：Kruskal（并查集）

```
struct DSU {
    vector<int> parent;
    DSU(int n) : parent(n) { iota(parent.begin(), parent.end(), 0); }
    int find(int x) { return parent[x] == x ? x : parent[x] = find(parent[x]); }
    void unite(int a, int b) { parent[find(a)] = find(b); }
};

int kruskal(int n, vector<array<int,3>> edges) { // {w,u,v}
    sort(edges.begin(), edges.end());
    DSU dsu(n);
    int mst_cost = 0, cnt = 0;
    for (auto [w, u, v] : edges) {
        if (dsu.find(u) != dsu.find(v)) {
            dsu.unite(u, v);
            mst_cost += w;
            if (++cnt == n-1) break;
        }
    }
    return cnt == n-1 ? mst_cost : -1; // -1 表示不连通
}
```

## 模板 9：Prim

```
int prim(vector<vector<pair<int,int>>>& adj, int n) {
    vector<bool> visited(n, false);
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    pq.push({0, 0}); // {weight, vertex}
    int mst_cost = 0, cnt = 0;
    while (!pq.empty()) {
        auto [w, u] = pq.top(); pq.pop();
        if (visited[u]) continue;
        visited[u] = true; mst_cost += w; cnt++;
        for (auto [v, weight] : adj[u])
            if (!visited[v]) pq.push({weight, v});
    }
    return cnt == n ? mst_cost : -1;
}
```

## 模板 10：快速幂（二分幂）

```
double fastPow(double x, int n) {
    double result = 1.0;
    long long N = abs((long long)n);
    while (N > 0) {
        if (N & 1) result *= x;
        x *= x;
        N >>= 1;
    }
    return n < 0 ? 1.0 / result : result;
}
```

## 模板 11：全排列（回溯）

```
vector<vector<int>> permute(vector<int>& nums) {
    vector<vector<int>> result;
    function<void(int)> backtrack = [&](int pos) {
        if (pos == nums.size()) { result.push_back(nums); return; }
        for (int i = pos; i < nums.size(); i++) {
            swap(nums[pos], nums[i]);
            backtrack(pos + 1);
            swap(nums[pos], nums[i]);     // 回溯恢复
        }
    };
    backtrack(0);
    return result;
}
```

## 模板 12：DFS 与 BFS

```
// DFS（递归版）
void dfs(int u, vector<bool>& vis, vector<vector<int>>& adj) {
    vis[u] = true;
    for (int v : adj[u]) if (!vis[v]) dfs(v, vis, adj);
}

// BFS
void bfs(int s, vector<vector<int>>& adj) {
    vector<bool> vis(adj.size(), false);
    queue<int> q; q.push(s); vis[s] = true;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v : adj[u]) if (!vis[v]) { vis[v] = true; q.push(v); }
    }
}
```

## 模板 13：拓扑排序（Kahn）

```
vector<int> topoSort(int n, vector<vector<int>>& adj) {
    vector<int> indegree(n, 0);
    for (int u = 0; u < n; u++)
        for (int v : adj[u]) indegree[v]++;
    queue<int> q;
    for (int i = 0; i < n; i++) if (indegree[i] == 0) q.push(i);
    vector<int> order;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        order.push_back(u);
        for (int v : adj[u])
            if (--indegree[v] == 0) q.push(v);
    }
    return order.size() == n ? order : vector<int>(); // 空表示有环
}
```

## 模板 14：最大子数组和（Kadane）

```
int maxSubArray(vector<int>& nums) {
    int cur = nums[0], best = nums[0];
    for (int i = 1; i < nums.size(); i++) {
        cur = max(nums[i], cur + nums[i]);
        best = max(best, cur);
    }
    return best;
}
```

## 模板 15：二分查找变体（lower_bound）

```
// 查找第一个 >= target 的位置
int lowerBound(vector<int>& A, int target) {
    int lo = 0, hi = A.size();
    while (lo < hi) {
        int mid = lo + (hi - lo) / 2;
        if (A[mid] >= target) hi = mid;
        else lo = mid + 1;
    }
    return lo;  // lo == 第一个 >= target 的位置
}
```
